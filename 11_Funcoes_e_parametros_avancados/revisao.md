    for linha in lista:
        if num in linha:
            return True 
    return False

def palindromo(palavra:str) -> bool:
    inverso = 0
    for i in len(palavra)
        inverso = inverso + palavra[i]
        i = i-1
    if palvra == inverso:
        return True 
    return False
